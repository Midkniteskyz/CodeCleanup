-- Count the number of monitored nodes from the ServerConfiguration table in Orion SCM
SELECT 
    Count(NodeID) AS [Monitored Node Count]  -- Column: Count of NodeID, renamed as Monitored Node Count

FROM 
    Orion.SCM.ServerConfiguration  -- Source table: ServerConfiguration in Orion SCM

WHERE 
    Enabled = 'True';  -- Filter criteria: Only count nodes where Enabled is 'True'
