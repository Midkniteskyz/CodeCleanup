SELECT 
    CASE 
        WHEN BuiltIn = 'True' THEN 'Out of the Box'
        ELSE 'Custom'
    END AS [Profile Type],  -- Column 1: Profile Type (Custom or Out of the Box)
    COUNT(*) AS [Count of Profiles]  -- Column 2: Count of Profiles
FROM 
    Orion.SCM.Profiles
WHERE 
    ManagedExternally != 'True'  -- Filter to exclude profiles managed externally
GROUP BY 
    CASE 
        WHEN BuiltIn = 'True' THEN 'Out of the Box'
        ELSE 'Custom'
    END;
