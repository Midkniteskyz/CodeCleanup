SELECT os.Hostname, os.ServerType, CASE WHEN os.Node.CPULoad IS NULL THEN 'Not found. Manually gather.' ELSE CONCAT(os.Node.CPULoad,'% of ', os.Node.CPUCount) END AS [CPU], CASE WHEN os.Node.PercentMemoryUsed IS NULL THEN 'Not found. Manually gather.' ELSE CONCAT(os.Node.PercentMemoryUsed,'% of ',ROUND(os.Node.TotalMemory/1024/1024/1024,0),' GB') END AS [Percent Memory Used] FROM Orion.OrionServers os
