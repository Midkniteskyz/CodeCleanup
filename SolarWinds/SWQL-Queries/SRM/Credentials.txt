SELECT 
    Name AS [Credential Display Name], 
    -- Extracting the Credential Type from the CredentialType field
    Substring(
        CredentialType, 
        -- Finding the starting position after 'Credentials.'
        CharIndex('Credentials.', CredentialType) + 12, 
        -- Calculating the length of the substring to extract
        (length(CredentialType) - CharIndex('.Credentials.', CredentialType))
    ) AS [Credential Type] 
FROM 
    Orion.Credential 
WHERE 
    credentialowner = 'SRM';
