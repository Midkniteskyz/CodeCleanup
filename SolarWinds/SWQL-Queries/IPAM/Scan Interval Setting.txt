-- Selecting group type text as Type, count of group IDs as Count, and formatted scan interval in hours
SELECT 
    Grouptypetext AS [Type],
    COUNT(GroupID) AS [Count],
    CONCAT(TOSTRING(ScanInterval / 60), ' hrs') AS [Scan Interval]
FROM 
    IPAM.GroupReport
WHERE 
    ScanInterval > 60  -- Filter to include only intervals greater than 60 seconds (1 minute)
    AND DisableAutoScanning <> 'true'  -- Exclude rows where auto scanning is disabled
GROUP BY 
    Grouptypetext, 
    ScanInterval;  -- Grouping by group type text and scan interval
